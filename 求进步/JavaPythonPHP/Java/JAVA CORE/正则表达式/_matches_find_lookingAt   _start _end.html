matches 匹配整个字符串：
find 找个模式匹配的字串； 二者会互相影响，需要reset（）方法来回复状态。     
          Pattern p = Pattern.compile("\\d{3,5}");
          String s = "123-34345-234-00";
          Matcher m = p.matcher(s);
          p(m.matches());          //false
          m.reset();               //回复，防止matches干扰下面find
          p(m.find());             //找字串从头开始找true
           p(m.start()+ "-" + m.end()); //输出找到的开头和结尾，必须找到，不然报错！！！！！
          p(m.find());             //true
          p(m.find());             //true
          p(m.find());             //false
          
          
          p(m.lookingAt());        // 从头开始找，true
          p(m.lookingAt());        // 还是从头开始找，true
          p(m.lookingAt());        // true
          p(m.lookingAt());        // true

     public static void p(Object o) {
          
          System.out.println(o);
     }
