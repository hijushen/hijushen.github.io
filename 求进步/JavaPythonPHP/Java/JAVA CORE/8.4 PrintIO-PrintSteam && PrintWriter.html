PrintStream 为其他输出流添加了功能，使它们能够方便地打印各种数据值表示形式。


import java.io.*;
public class TestPrintStream1 {
  public static void main(String[] args) {
    PrintStream ps = null;  //新建一个PrintStream为null的对象
    try {
      FileOutputStream fos =
              new FileOutputStream("d:\\bak\\log.dat");  //确定要记录数据的文件，Output针对程序而言，故为Output，注意2 个斜杠，一个为转译。
      ps = new PrintStream(fos);//在程序输出流外加一个PrintSteam的管子
    } catch (IOException e) {
      e.printStackTrace();
    }
    if(ps != null){
      System.setOut(ps);   //重新设置输出形式
    }
    int ln = 0;            //换行变量
    for(char c = 0; c <= 60000; c++){
      System.out.print(c+ " ");  // c为char变量，数字会强制转换。
      if(ln++ >=100){ System.out.println(); ln = 0;} //换行控制
    }
  }
}

import java.io.*;
public class TestPrintStream2 {  //将输入的文件名对应的文件内容读出
  public static void main(String[] args) {
    String filename = args[0];  //将文件名放入数组中，进行传递
    if(filename!=null){list(filename,System.out);} //有文件名，传递参数至list方法中
  }
  public static void list(String f,PrintStream fs){
    try {
      BufferedReader br =
                  new BufferedReader(new FileReader(f));
      String s = null;
      while((s=br.readLine())!=null){
        fs.println(s);           
      }
      br.close();
    } catch (IOException e) {
      fs.println("无法读取文件");
    }
  }
}



import java.util.*;
import java.io.*;
public class TestPrintStream3 {//记录输入的记录，日志和日期。
  public static void main(String[] args) {
    String s = null;
    BufferedReader br = new BufferedReader(
                        new InputStreamReader(System.in));
    try {
      FileWriter fw = new FileWriter
                           ("d:\\bak\\logfile.log", true); //Log4J
      PrintWriter log = new PrintWriter(fw);
      while ((s = br.readLine())!=null) {
        if(s.equalsIgnoreCase("exit")) break;
        System.out.println(s.toUpperCase());
        log.println("-----");
        log.println(s.toUpperCase());
        log.flush();
      }
      log.println("==="+new Date()+"===");
      log.flush();
      log.close();
    } catch (IOException e) {
      e.printStackTrace();
    }
  }
}
