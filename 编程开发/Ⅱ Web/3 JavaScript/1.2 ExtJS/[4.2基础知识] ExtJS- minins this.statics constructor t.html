2. 多态方法statics. constructor. 主要this.statics()与this.self的区别.
statics: 类的静态方法清单.
constructor构造执行.对应结果如下.
statics与self(动态)的区别
  1. Ext.define('My.Cat', {
  2.     statics: {
  3.         totalCreated: 0,
  4.         speciesName: 'Cat'
  5.     },
  6.     constructor: function () {
  7.         var statics = this.statics();//获得静态清单.
  8.         alert(statics.speciesName); //静态名称
  9.         alert(this.self.speciesName);//真正对象实例名
  10.         statics.totalCreated++;
  11.     },
  12.     clone: function () {
  13.         var cloned = new this.self; //实际对象的克隆,new产生构造
  14.         cloned.groupName = this.statics().speciesName; //赋值新属性值
  15.         return cloned;//返回对象
  16.     }
  17. });
  19. Ext.define('My.SnowLeopard', {
  20.     extend: 'My.Cat',
  21.     statics: {
  22.         speciesName: 'Snow Leopard'
  23.     },
  24.     constructor: function () {
  25.         this.callParent();//使用父类同名方法:constructor
  26.     }
  27. });
  29. var cat = new My.Cat();//执行构造,为cat cat
  30. var snowLeopard = new My.SnowLeopard();//静态为cat,覆写的实例为snow leopard. 所有结果为cat / snow leopard
  31. var clone = snowLeopard.clone();//实际被克隆,静态仍然为cat. 所以为cat / snow leopard
  32. alert(Ext.getClassName(clone)); //实际类名: My.SnowLeopard
  33. alert(My.Cat.totalCreated); //3
  34. alert(clone.groupName);//cat,静态值.

initConfig, 类的初始化配置.

  9.     }
  10. });
