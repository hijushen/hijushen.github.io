1. 无法访问路径下local:端口/data/users.json.

  1. proxy: {
  2.         type: 'ajax',
  3.         api: {
  4.             read:"app/data/users.html",
  5.             update: "app/data/updateUsers.html"
  6.         },
  7.         
  8.         reader: {
  9.             type: 'json',
  10.             root: 'users',
  11.             successProperty:'success'
  12.         }
  13.     }
  14. });
