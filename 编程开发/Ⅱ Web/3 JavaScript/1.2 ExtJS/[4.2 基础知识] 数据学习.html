1. 最简单的使用Store, 静态数据绑定ComboBox的模式:
  1. var states = Ext.create('Ext.data.Store', {
  2.     fields: ['abbr', 'name'],
  3.     data: [
  4.         { "abbr": "dawang", "name": "xunshang" },
  5.         { "abbr": "erwang", "name": "chifan" },
  6.         { "abbr": "goup", "name": "yoyo" }
  7.     ]
  8. });
  10. Ext.create('Ext.form.ComboBox', {
  11.     fieldLabel: 'Choose State',
  12.     store: states,
  13.     queryMode: 'local',
  14.     displayField: 'name',
  15.     valueField: 'abbr',
  16.     renderTo: Ext.getBody()
  17. });
