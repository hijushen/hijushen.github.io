javascript数组: 
--push pop. 数组尾部压入一个/多个元素,结果为数量,数组数量变更.  弹出数组尾部弹出一个元素
--shift unshift. 删除[0],返回[0]; 添加一个/多个元素到数组开头,返回新数组长度. 数组元素内容变更.
--splice slice . splice(index,length,item1,...): 删除指定index开始length长度,使用item1开始的数组添加替代. 改变数组; slice(stard,end) 获得已有数组指定索引start->end的子数组. 不指定end到尾或者头. 不改变原有数组.
--concat join//连接字符串a.concat(b,c)--三个连接. arr.join(separator)返回字符串. separate不指定默认为,
--sort reverse 排序 / 倒置数组. 其中排序默认按照字母, 如果数字排序需要指定排序方法
--ECMA5拓展了很多:
  foreach:
  var a = [999,1, 2, 3, 45,99,778];
    a.forEach(function (item) {
        alert(item);
    });

  filter:
  var newArr = a.filter(function (item) {
        if (item > 20) {
            return true;
        } else {
            return false;
        }
    });

  alert(newArr); //999,45,99,778
  alert(typeof newArr); //object
  alert(newArr[0]); //999. 说明newArr还是数组

    function sortable(a, b) {
        return a - b;
    }

    var a = [999,1, 2, 3, 45,99,778];    
    a.sort();
    alert(a); //1,2,3,45,778,99,999. 默认按照字母排列
    a.sort(sortable);//加上排序算法,排序大小
    alert(a);
