1. 总结任何一个AutoResetEvent调用WaitOne()时阻塞，等待相应Event的Set()指令.

  1.         public static AutoResetEvent _workerEvent = new AutoResetEvent(false);
  2.         private static AutoResetEvent _mainEvent = new AutoResetEvent(false);
  4.         static void Main(string[] args)
  5.         {
  6.             var t = new Thread(() => Process(10));
  7.             t.Start();
  9.             Console.WriteLine("Waiting for another thread to complete work");
  10.             _workerEvent.WaitOne();
  11.             Console.WriteLine("First operation is completed");
  12.             Console.WriteLine("Performing an operation on a main thread");
  13.             Thread.Sleep(TimeSpan.FromSeconds(5));
  14.             _mainEvent.Set();
  15.             Console.WriteLine("Nw running the second operation on a second thread");
  16.             _workerEvent.WaitOne();
  17.             Console.WriteLine("Second operation is completed");
  18.         }
  20.         static void Process(int seconds)
  21.         {
  22.             Console.WriteLine("Starting a long running work ...");
  23.             Thread.Sleep(TimeSpan.FromSeconds(seconds));
  24.             Console.WriteLine("Work is done!");
  26.             _workerEvent.Set();
  27.             Console.WriteLine("Waiting for a main thread to complete its work");
  28.             _mainEvent.WaitOne();
  29.             Console.WriteLine("Starting second operation...");
  31.             Thread.Sleep(TimeSpan.FromSeconds(seconds));
  32.             Console.WriteLine("Work is done!");
  33.             _workerEvent.Set();
  34.         }
