  2.     {
  3.         public T data;
  5.         private Node<T> leftNode;
  6.         public Node<T> LeftNode { get; set; }
  8.         private Node<T> rightNode;
  9.         public Node<T> RightNode { get; set; }
  11.         public Node(T data)
  12.         {
  13.             this.data = data;
  14.         }
  16.         public override string ToString()
  17.         {
  18.             return data.ToString();
  19.         }
  20.     }
  22.     public class BinaryTreeNode<T> where T:IComparable<T>
  23.     {
  24.         private Node<T> root;
  25.         public Node<T> Root { get; set; }
  27.         public void Add(T item)
  28.         {
  29.             if(root == null)
  30.             {
  31.                 root = new Node<T>(item);
  32.                 return;
  33.             }
  34.             Add(item, root);
  35.         }
  37.         private void Add(T item, Node<T> node)
  38.         {
  39.             if (item.CompareTo(node.data) < 0)
  40.             {
  41.                 if (node.LeftNode == null)
  42.                     node.LeftNode = new Node<T>(item);
  43.                 else
  44.                     Add(item, node.LeftNode);
  45.             }
  46.             else if (item.CompareTo(node.data) > 0)
  47.             {
  48.                 if (node.RightNode == null)
  49.                     node.RightNode = new Node<T>(item);
  50.                 else
  51.                 {
  52.                     Add(item, node.RightNode);
  53.                 }
  54.             }
  55.         }
  57.         public void AddRange(params T[] items)
  58.         {
  59.             foreach (var item in items)
  60.             {
  61.                 Add(item);
  62.             }
  63.         }
  65.         public void Print()
  66.         {
  67.             Print(root, 0, Console.WindowWidth / 2);
  68.         }
  70.         public IEnumerable<T> Inorder
  71.         {
  72.             get { return GetInOrder(this.root); }
  73.         }
  75.         int depth = 0;
  77.         /// <summary>
  78.         /// 中序遍历: 左 根 右
  79.         /// </summary>
  80.         /// <param name="node"></param>
  81.         /// <returns></returns>
  82.         private IEnumerable<T> GetInOrder(Node<T> node)
  83.         {
  84.             if (depth++ > 5) yield break;
  85.             
  86.             if(node.LeftNode != null)
  87.             {
  88.                 foreach (T item in GetInOrder(node.LeftNode))
  89.                 {
  90.                     yield return item;
  91.                 }
  92.             }
  94.             yield return node.data;
  96.             if(node.RightNode !=null)
  97.             {
  98.                 foreach (T item in GetInOrder(node.RightNode))
  99.                 {
  100.                     yield return item;
  101.                 }
  102.             }
  103.         }
  104.         private void Print(Node<T> item, int depth, int offset)
  105.         {
  108.             if (item == null) return;
  109.             Console.CursorLeft = offset;
  110.             Console.CursorTop = depth;
  111.             Console.Write(item.data);
  113.             if (item.LeftNode != null)
  114.                 Print("/", depth + 1, offset - 1);
  115.             Print(item.LeftNode, depth + 2, offset - 3);
  117.             if (item.RightNode != null)
  118.                 Print("\\", depth + 1, offset + 1);
  119.             Print(item.RightNode, depth + 2, offset + 3);
  120.         }
  122.         private void Print(string s,int depth,int offset)
  123.         {
  124.             Console.CursorLeft = offset;
  125.             Console.CursorTop = depth;
  126.             Console.Write(s);
  127.         }
  128.     }

- Predicate<T> 接受一个由T指定的参数并返回一个布尔值. 用于 `Array.Find`和`Array.FindAll`这样的函数中, `Predicate<T>`可以同过普通函数,匿名委托或者`lambda`表达式来初始化. 完整代码示例:
```

  8.         {

  10.             Predicate<Rectangle> area = 
  11.                 rect =>
  12.                     (rect.Width * rect.Height) <= (random.Next(200) *
  13.                     random.Next(200));
  15.             Rectangle[] matches = Array.FindAll(rects, area);
  16.             e.Graphics.Clear(this.BackColor);
  18.             foreach (var rect in matches)
  19.             {
  20.                 e.Graphics.DrawRectangle(Pens.Red, rect);
  21.                 e.Graphics.DrawString("Found: " + matches.Length.ToString(),
  22.                     this.Font, Brushes.Black, 10, ClientRectangle.Height - 40);
  24.             }
  25.         }

```

- 柯里化
将一个多步骤表达式拆分成多个`lambda`表达式, 每个`lambda`表达式都是下一个`lambda`表达式的资源提供者. 代码, 同样的功能2种方式:
```
  1.             Func<int,int,int> longLambda = (x, y) => x + y;
  2.             Console.WriteLine(longLambda(3,4));
  3.             Console.ReadLine();
  5.             Func<int,Func<int,int>> curry1 = x => y => x + y;//柯里化: 与上一个同样的功能.
  6.             Console.WriteLine(curry1(3)(4));
  7.             Console.ReadLine();
```
# 总结
- `Lambda`表达式就是匿名委托的简洁表示法, 是内联函数. Lambda表达式返回匿名委托(如Func<T> Predicate<T>以及Action<T>等)的实例. **这些委托在使用函数来处理数组或其他可枚举类型的拓展方法中非常有用**
- 其他主题: 投影(Select), 柯里化以及闭包.
