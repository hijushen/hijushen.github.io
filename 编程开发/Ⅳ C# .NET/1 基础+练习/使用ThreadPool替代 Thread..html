使用下面的地方,替代方案:
使用前:
Thread t = new Thread(() =>
{
    while (true)
    {
        if (cts.IsCancellationRequested)
        {
            Console.WriteLine("线程被取消");
            break;
        }
 
        Console.WriteLine(DateTime.Now.ToString());
        Thread.Sleep(1000);
    }
});
 
t.Start();
替代方案:
ThreadPool.QueueUserWorkItem((objState) =>  //object为WaitCallBack要执行的方法,不能少
                {
                    while(true)
                    {
                        if (cts.IsCancellationRequested)
                        {
                            Console.WriteLine("线程被取消");
                            break;
                        }
 
                        Console.WriteLine(DateTime.Now.ToString());
                        Thread.Sleep(1000);
                    }
                },null); //null可删除有重载,代表方法是用数据object
