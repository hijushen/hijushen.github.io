static void Main(string[] args)
        {
            Task t = new Task(() => //可以使用delegata() 使用
            {
                Console.WriteLine("任务开始工作...");
                Thread.Sleep(5000);
            });
            t.Start();
            t.ContinueWith((task) => // 可以使用delegate(Task task) 使用         
                {
                    Console.WriteLine("作业完成,完成时的状态为: ");
                    Console.WriteLine("IsCancel={0}\tIsCompleted={1}\tIsFaulted={2}",
                        t.IsCanceled, t.IsCompleted, t.IsFaulted);
                });
            Console.ReadKey();
        }
static void Main(string[] args)
        {
            #region 复杂的Task支持完成通知\取消\获取任务返回值
 
            CancellationTokenSource cts = new CancellationTokenSource();
            Task<int> t = new Task<int>(() => Add(cts.Token), cts.Token);
            t.Start();
            t.ContinueWith(TaskEnded);
 
            Console.ReadKey();
            cts.Cancel();
            Console.ReadKey();
            #endregion
        }
        static void TaskEnded(Task<int> task)
        {
            Console.WriteLine("任务完成, 完成时的状态为: ");
            Console.WriteLine("IsCanceled={0}\tIsCompleted={1}\tIsFaulted={2}",task.IsCanceled,
                task.IsCompleted,task.IsFaulted);
            Console.WriteLine("任务返回值为: {0}", task.Result);
        }
 
        static int Add(CancellationToken ct)
        {
            Console.WriteLine("任务开始... ....");
            int result = 0;
            while(!ct.IsCancellationRequested)
            {
                result++; 
                Thread.Sleep(1000);
            }
            return result;
        }
