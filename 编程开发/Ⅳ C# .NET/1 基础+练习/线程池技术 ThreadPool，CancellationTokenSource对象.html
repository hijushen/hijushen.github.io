  23.             }

2. 协作式取消线程池线程
  1.         static void Main(string[] args)
  2.         {
  3.             Console.WriteLine("Main thread run");
  4.             CancellationTokenSource cts = new CancellationTokenSource(); //1.声明一个对象
  5.             ThreadPool.QueueUserWorkItem(callBack,cts.Token);2.//传入值
  6.             Console.WriteLine("press enter to cancel !");
  7.             Console.Read();
  8.             cts.Cancel(); //取消请求
  10.             Console.ReadKey();
  11.         }
  13.         static void callBack(object state)
  14.         {
  15.             CancellationToken token = (CancellationToken)state;
  16.             Console.WriteLine("begin count");
  17.             Count(token, 1000);
  18.         }
  20.         static void Count(CancellationToken token, int count)
  21.         {
  22.             for (int i = 0; i < count; i++)
  23.             {
  24.                 if(token.IsCancellationRequested)
  25.                 {
  26.                     Console.WriteLine("count cancel");
  27.                     return;
  28.                 }
  29.                 Console.WriteLine("count:"+i);
  30.                 Thread.Sleep(300);
  31.             }
  32.             Console.WriteLine("count done!");
  33.         }
