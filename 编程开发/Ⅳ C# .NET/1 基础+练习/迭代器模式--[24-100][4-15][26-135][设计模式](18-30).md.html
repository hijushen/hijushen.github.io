### 在1.0 中, 没有`yield return` 关键字需要如下的实现方式：MoveNext()是延迟lazy 加载的。
```

        public IEnumerator GetEnumerator()
        {
            return new FriendIterator(this);
        }  
```
重新定义一个迭代器类，用来描述：
```
  1. /// <summary>
  2.     ///  自定义迭代器，必须实现 IEnumerator接口
  3.     /// </summary>
  4.     public class FriendIterator : IEnumerator
  5.     {
  6.         private readonly Friends friends;
  7.         private int index;
  8.         private Friend current;
  9.         internal FriendIterator(Friends friendcollection)
  10.         {
  11.             this.friends = friendcollection;
  12.             index = 0;
  13.         }
  15.         #region 实现IEnumerator接口中的方法
  16.         public object Current
  17.         {
  18.             get
  19.             {
  20.                 return this.current;
  21.             }
  22.         }
  24.         public bool MoveNext()
  25.         {
  26.             if (index + 1 > friends.Count)
  27.             {
  28.                 return false;
  29.             }
  30.             else
  31.             {
  32.                 this.current = friends[index];
  33.                 index++;
  34.                 return true;
  35.             }
  36.         }
  38.         public void Reset()
  39.         {
  40.             index = 0;
  41.         }
  43.         #endregion
```

## 一般化的迭代器模式--[24/100][4/15][26/135][设计模式](18/30) (2015-06-29 17:34:28)

```
  1.     class Program
  2.     {
  3.         static void Main(string[] args)
  4.         {
  5.             Iterator iterator;
  6.             IListCollection list = new ConcreteList();
  8.             iterator = list.GetIterator();
  10.             while(iterator.MoveNext())
  11.             {
  12.                 int i = (int)iterator.GetCurrent();
  13.                 Console.WriteLine(i.ToString());
  14.                 iterator.Next();
  15.             }
  17.             Console.ReadKey();
  18.         }
  19.     }
  21.     public interface IListCollection
  22.     {
  23.         Iterator GetIterator();
  24.     }
  26.     public interface Iterator
  27.     {
  28.         bool MoveNext();
  29.         Object GetCurrent();
  30.         void Next();
  31.         void Reset();
  32.     }
  34.     public class ConcreteIterator:Iterator
  35.     {
  36.         private ConcreteList _List;
  37.         private int _index;
  39.         public ConcreteIterator(ConcreteList list)
  40.         {
  41.             _List = list;
  42.             _index = 0;
  43.         }
  45.         public bool MoveNext()
  46.         {
  47.             if (_index < _List.Length)
  48.             { return true; }
  49.             return false;
  50.         }
  52.         public Object GetCurrent()
  53.         {
  54.             return _List.GetElement(_index);
  55.         }
  57.         public void Reset()
  58.         {
  59.             _index = 0;
  60.         }
  62.         public void Next()
  63.         {
  64.             if (_index < _List.Length)
  65.             {
  66.                 _index++;
  67.             }
  68.         }
  70.     }
  72.     public class ConcreteList:IListCollection
  73.     {
  74.         int[] collection;
  75.         public ConcreteList()
  76.         {
  77.             collection = new int[] { 2, 4, 6, 8 };
  78.         }
  80.         public Iterator GetIterator()
  81.         {
  82.             return new ConcreteIterator(this);
  83.         }
  85.         public int Length
  86.         {
  87.             get { return collection.Length; }
  88.         }
  90.         public int GetElement(int index)
  91.         {
  92.             return collection[index];
  93.         }
  94.     }
```
