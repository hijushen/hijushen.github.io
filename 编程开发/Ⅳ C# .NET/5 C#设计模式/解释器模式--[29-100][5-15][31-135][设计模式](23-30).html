  1.     class PlayContext
  2.     {
  3.         private string text;
  4.         public string PlayText
  5.         {
  6.             get { return text; }
  7.             set {text = value;}
  8.         }
  9.     }
  11.     abstract class Expression
  12.     {
  13.         //解释器
  14.         public void  Interpret(PlayContext context) //1--割context分段; 剩下往下; 2--已割下的执行本文虚方法,后续执行.
  15.         {
  16.             if (context.PlayText.Length ==0)
  17.             {
  18.                 return;
  19.             }
  20.             else
  21.             {
  22.                 string playKey = context.PlayText.Substring(0, 1);
  23.                 context.PlayText = context.PlayText.Substring(2);
  25.                 double playValue = Convert.ToDouble(context.PlayText.Substring(0, context.PlayText.IndexOf(" ")));
  26.                 context.PlayText = context.PlayText.Substring(context.PlayText.IndexOf(" ") + 1);
  28.                 Excute(playKey, playValue);
  29.             }
  30.         }
  32.         public abstract void Excute(string key, double value); //钩子, 将方法执行后续延续
  33.     }
  35.     //音符类
  36.     class Note:Expression
  37.     {
  38.         public override void Excute(string key, double value)
  39.         {
  40.             string note = "";
  41.             switch(key)
  42.             {
  43.                 case "C":
  44.                     note = "1";
  45.                     break;
  46.                 case "D":
  47.                     note = "2";
  48.                     break;
  49.                 case "E":
  50.                     note = "3";
  51.                     break;
  52.                 case "F":
  53.                     note = "4";
  54.                     break;
  55.                 case "G":
  56.                     note = "5";
  57.                     break;
  58.                 case "A":
  59.                     note = "6";
  60.                     break;
  61.                 case "B":
  62.                     note = "7";
  63.                     break;                    
  64.             }
  65.             Console.Write("{0}", note);
  66.         }
  67.     }
  70.     class Scale:Expression
  71.     {
  72.         public override void Excute(string key, double value)
  73.         {
  74.             string scale = "";
  75.             switch (Convert.ToInt32(value))
  76.             {
  77.                 case 1:
  78.                     scale = "低音";

  89.     }
