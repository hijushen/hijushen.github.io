1. 适用于: 功能比较复杂,需要维护或记录属性历史记录的类, 或者需要保存的属性只是众多属性中的一小部分, Originator 可以根据保存的memento信息还原到前一状态.
缺点 : 耗费内存.
代码:
  1.     class Program
  2.     {
  3.         static void Main(string[] args)
  4.         {
  5.             Originator o = new Originator(); //原始类
  6.             o.State = "ON";
  7.             o.Show();
  9.             CareTaker c = new CareTaker(); //管理类
  10.             c.Memento = o.CreateMemento();//快照来了.
  12.             o.State = "OFF";
  13.             o.Show();
  15.             o.SetMemento(c.Memento);//还原快照.
  16.             o.Show();
  18.             Console.Read();
  19.         }
  20.     }
  1.     class Originator
  2.     {
  3.         private string state;
  5.         public string State
  6.         {
  7.             get { return state; }
  8.             set { state = value; }
  9.         }
  11.         public Memento CreateMemento()
  12.         {
  13.             return (new Memento(state));
  14.         }
  16.         public void SetMemento(Memento memento)
  17.         {
  18.             state = memento.State;
  19.         }
  21.         public void Show()
  22.         {
  23.             Console.WriteLine("State = " + state); //显示数据
  24.         }
  25.         
  26.     }
  28.     
  29.     //备忘录
  30.     class Memento
  31.     {
  32.         private string state;
  34.         public string State
  35.         {
  36.             get { return state; }
  37.         }
  38.         public Memento(string state) //构造方法,相关数据导入
  39.         {
  40.             this.state = state;
  41.         }
  42.     }
  44.     //管理者
  45.     class CareTaker
  46.     {
  47.         private Memento memento;
  48.         public Memento Memento
  49.         {
  50.             get
  51.             {
  52.                 return memento;
  53.             }
  54.             set
  55.             {
  56.                 memento = value;
  57.             }
  58.         }
  59.     }
