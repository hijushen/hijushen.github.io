Handler (抽象处理者)
ConcreteHandler (具体处理者)
具体案例:
       总价大于1万小于2万5的则还需要副总进行批准，总价大于2万5小于10万的需要还需要总经理批准，而大于总价大于10万的则需要组织一个会议进行讨论。对于这样一个需求，最直观的方法就是设计一个方法，参数是采购的总价，然后在这个方法内对价格进行调整判断，然后针对不同的条件交给不同级别的人去处理，这样确实可以解决问题，但这样一来，我们就需要多重if-else语句来进行判断，但当加入一个新的条件范围时，我们又不得不去修改原来设计的方法来再添加一个条件判断，这样的设计显然违背了“开-闭”原则。这时候，可以采用责任链模式来解决这样的问题。具体实现代码如下所示。
  1.             PurchaseRequest requestTel = new PurchaseRequest(10000, "Telphone");
  2.             PurchaseRequest requestSoft = new PurchaseRequest(40000, "Telphone");
  3.             PurchaseRequest requestComputer = new PurchaseRequest(110000, "Telphone");
  5.             Approver manger = new Manager("2B");
  6.             Approver VP = new VicePresident("SB");
  7.             Approver presicent = new President("DSB");
  9.             //设置责任链
  10.             manger.NextApprover = VP;
  11.             VP.NextApprover = presicent;
  13.             //处理请求
  14.             manger.ProcessRequest(requestTel);
  15.             manger.ProcessRequest(requestSoft);
  16.             manger.ProcessRequest(requestComputer);
  18.             Console.ReadLine();
  1.     public class PurchaseRequest
  2.     {

  7.

  9.         {
  10.             Amount = amount;
  11.             ProductName = productName;
  12.         }
  13.     }
  1.     public abstract class Approver
  2.     {
  3.         public Approver NextApprover { get; set; }
  5.         public string Name { get; set; }
  7.         public Approver(string name)
  8.         {
  9.             this.Name = name;
  10.         }
  12.         public abstract void ProcessRequest(PurchaseRequest request);
  13.     }
  15.     public class Manager:Approver
  16.     {
  17.         public Manager(string name):base(name)
  18.         {
  20.         }
  22.         public override void ProcessRequest(PurchaseRequest request)
  23.         {

  25.             {
  26.                 Console.WriteLine("{0} - {1} approved the request of purshing {2}", this,      Name, request.ProductName);
  27.             }
  28.             else if (NextApprover != null)
  29.             {
  30.                 NextApprover.ProcessRequest(request);
  31.             }
  32.         }
  33.     }
  35.     public class VicePresident:Approver
  36.     {

  47.                 NextApprover.ProcessRequest(request);

  49.         }
  50.     }
  52.     public class President : Approver
