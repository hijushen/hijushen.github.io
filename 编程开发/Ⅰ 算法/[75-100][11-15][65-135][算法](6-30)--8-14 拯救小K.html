拯救小K的思路
1, 输入地图, n * m 的地图.
2, 输入起点重点横竖坐标. startx, starty, p,q,.
3, dfs深度优先遍历.
    a) 边界条件: x=p,y=q到达位置. 判定min是否最小.是, 则保存, 返回. 
    b) 二维数组保存走路方式. 
    c) 超出边界, 重新continue.
    d) 循环四种走路方式. 判定无障碍和未走过则更新标记.
       e) 走step +1步.
       f) 收回step+1步.
     g) 返回.

  1. #include "stdio.h"
  3. int n, m, p, q, min = 9999;
  4. int a[51][51], book[51][51];
  6. void dfs(int x, int y, int step)
  7. {
  8. 	int  next[4][2] = { { 0, 1 }, //向上
  9. 								{ 1, 0 }, //向右
  10. 								{0,-1}, //向下
  11. 								{-1,0} //向左
  12. 								};
  14. 	int tx, ty, k;
  16. 	//判断是否到达位置
  17. 	if (x == p, y == q)
  18. 	{
  19. 		if (step < min)
  20. 		{
  21. 			min = step;
  22. 		}
  23. 		return;
  24. 	}
  26. 	for (k = 0; k <= 3; k++)
  27. 	{
  28. 		tx = x + next[k][0];
  29. 		ty = y + next[k][1];
  31. 		if (tx<1 || tx>n || ty<1 || ty>n) continue;
  33. 		if (a[tx][ty] == 0 && book[tx][ty] == 0)
  34. 		{
  35. 			book[tx][ty] = 1;
  36. 			dfs(tx, ty, step + 1);
  37. 			book[tx][ty] = 0;
  38. 		}
  39. 	}
  41. 	return;
  42. }
  44. int main()
  45. {
  46. 	int i, j, startx, starty;
  48. 	scanf_s("%d %d", &n, &m);
  50. 	for (i = 1; i <= n; i++)
  51. 		for (j = 1; j <= m; j++)
  52. 		{
  53. 			scanf_s("&d", &a[i][j]);
  54. 		}
  56. 	scanf_s("%d %d %d %d", &startx, &starty, &p, &q);
  58. 	book[startx][starty] = 1;
  59. 	dfs(startx, starty, 0);
  61. 	printf_s("%d", min);
  62. 	getchar(); getchar();
  63. 	return 0;
  64. }
