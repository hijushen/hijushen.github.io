 public ExecutionResult QueryType(string location, string partno)
        {
            ExecutionResult res;
            DataSet ds = new DataSet();
            res = tSql.LocationType(location); ///
            ds = ( DataSet)res.Anything;
            string LocationType = ds.Tables[0].Rows[0][0].ToString();
            res = tSql.PartType(partno);
            ds = ( DataSet)res.Anything;
            string PartType = ds.Tables[0].Rows[0][0].ToString();
            if (Convert .ToInt32(PartType)==2)
            {

                PartType = "SMALL";
            }
            else
            {
                PartType = "LARGE";  
            }
            if (PartType==LocationType)
            {
                res.Status = true;
            }
            else
            {
                res.Status = false;  
            }
            return res;
